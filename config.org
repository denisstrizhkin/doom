** openwith

#+BEGIN_SRC emacs-lisp :results none
(use-package! openwith
  :config
  (setq openwith-associations (list
       (list (openwith-make-extension-regexp
              '("m4v" "mp4" "MP4" "MTS" "mpg" "mov" "avi" "AVI" "flv" "mkv" "webm"))
             "mpv"
             '(file))
       (list (openwith-make-extension-regexp
              '("pdf"))
             "zathura"
             '(file))))
  (openwith-mode 1))
#+END_SRC
